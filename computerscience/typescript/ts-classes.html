üìë Inheritance (Extending Classes)

class Admin extends User {
  role: string;
  constructor(name: string, role: string) {
    super(name);
    this.role = role;
  }

  showRole(): string {
    return this.name + " is an " + this.role;
  }
}

const admin1 = new Admin("Jamie", "Administrator");
console.log(admin1.showRole());
